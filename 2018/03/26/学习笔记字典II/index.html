#1.创建字典

user_0 = {

‘username’ : ‘fancunjun’,

‘first_name’:’fan’,

‘last_name’:’cunjun’

}

#2.遍历字典

for key,value in user_0.items():

print(‘\nkey:’+ key,’value:’+value)

#3.遍历字典中的所有的键

for name in user_0:

print(name)

#4.给出指定朋友喜欢的编程语言

favorite_langeages = {

‘join’:’c’,

‘jack’:’java’,

‘mike’:’c++’,

‘fancunjun’:’python’

}

friends = [‘jack’,’fancunjun’]

for name in favorite_langeages :

if name in friends:

print(‘hi ,’+ name + ‘i see you favorite_langeages is ‘ + favorite_langeages[name])

#6.按照顺序遍历字典

favorite_langeages = {

‘join’:’c’,

‘jack’:’java’,

‘mike’:’c++’,

‘fancunjun’:’python’

}

for name in sorted(favorite_langeages.keys()):

print(“thank you!” + name.title())

#7.遍历字典中所有的值

#favorite_langeages = {

‘fancunjun’:’python’

}

print(‘以下是打印字典中所的值’)

for va in favorite_langeages.values():

print(va.title())

#8.利用set去除字典中的重复项

favorite_langeages = {

‘join’:’c’,

‘jacd’:’python’,

‘mike’:’c++’,

‘fancunjun’:’python’

}

print(‘以下是利用set集合去除重复项’)

for val in set(favorite_langeages.values()):

print(val.title())

#9.编写词汇表

glossary = {

‘变量’:’是可改变，可以通过变量名来访问变量’,

‘列表’:’存储有序的元素的组合’,

‘元组’:’存储元素，不可被更改’,

‘字典’:’存储一对键值对’,

‘索引’:’元素的下标位置，可通过索引来访问元素’,

‘vscode’:’配置python环境比较难’

}

print(‘打印字典的键值对’)

for keys,values in glossary.items():

print(keys,values)

}
